# Hyprland Configuration Variables Reference
# Quick reference for common Hyprland configuration options

## GENERAL
general:gaps_in          Inner gaps between windows (default: 6)
general:gaps_out         Outer gaps to screen edges (default: 10)
general:border_size      Window border thickness (default: 2)
general:col.active_border   Color of active window border
general:col.inactive_border Color of inactive window border
general:resize_on_border    Allow resize by dragging borders (true/false)
general:layout           Window layout manager (dwindle/master)
general:allow_tearing    Enable tearing for performance (true/false)

## DECORATION
decoration:rounding      Corner radius for windows (default: 8)
decoration:shadow:enabled       Enable window shadows (true/false)
decoration:shadow:range         Shadow distance (default: 2)
decoration:shadow:render_power  Shadow blur intensity (default: 3)
decoration:blur:enabled   Enable background blur (true/false)
decoration:blur:size      Blur radius (default: 6)
decoration:blur:passes    Blur quality passes (default: 2)

## ANIMATIONS
animations:enabled       Enable animations (true/false)
animations:bezier        Define custom bezier curves
animations:animation     Configure specific animation types

## INPUT
input:kb_layout          Keyboard layout (us, es, fr, etc)
input:kb_variant         Keyboard variant
input:kb_options         XKB options (e.g., compose:caps)
input:repeat_rate        Key repeat rate in Hz (default: 40)
input:repeat_delay       Key repeat delay in ms (default: 600)
input:follow_mouse       Focus follows mouse (0=no, 1=yes, 2=always)
input:sensitivity        Mouse sensitivity (-1.0 to 1.0)
input:touchpad:natural_scroll      Natural scrolling (true/false)
input:touchpad:tap-and-drag        Tap and drag (true/false)
input:gestures:workspace_swipe     Workspace swipe (true/false)

## DWINDLE LAYOUT
dwindle:pseudotile       Pseudotile windows (true/false)
dwindle:preserve_split   Preserve split direction (true/false)
dwindle:force_split      Force split direction (0=auto, 1=horiz, 2=vert)

## MASTER LAYOUT
master:new_status        Where to place new windows (master/stack)
master:new_on_top        New windows on top (true/false)
master:mfact             Master area factor (0.0-1.0)

## MISC
misc:disable_hyprland_logo      Hide logo on startup (true/false)
misc:disable_splash_rendering   Hide splash screen (true/false)
misc:focus_on_activate          Auto-focus new windows (true/false)
misc:enable_swallow             Window swallowing (true/false)

## CURSOR
cursor:hide_on_key_press        Hide cursor when typing (true/false)
cursor:no_hardware_cursors      Disable hardware cursors (true/false)

## MONITOR SYNTAX
monitor = NAME, RESOLUTION, POSITION, SCALE
# NAME: Monitor name (DP-1, HDMI-A-1, eDP-1) or "," for auto
# RESOLUTION: WIDTHxHEIGHT@REFRESH or "preferred"
# POSITION: X,Y coordinates or "auto"
# SCALE: Scaling factor (1, 1.5, 2, etc)

## WINDOW RULES
windowrule = RULE, CLASS
windowrulev2 = RULE, CONDITIONS

# Common rules:
# float, tile, fullscreen, maximize
# opacity ACTIVE INACTIVE
# workspace ID
# size WIDTH HEIGHT
# center
# noblur, noborder, noshadow

## LAYER RULES
layerrule = RULE, NAMESPACE
# Rules: blur, noanim, dimaround

## EXEC COMMANDS
exec = COMMAND               # Execute on config reload
exec-once = COMMAND          # Execute once at startup

## BIND SYNTAX
bind = MODIFIERS, KEY, DISPATCHER, PARAMS
bindd = MODIFIERS, KEY, DESCRIPTION, DISPATCHER, PARAMS
bindl = ... (release)
bindr = ... (repeat)
binde = ... (repeat when held)
bindn = ... (non-consuming)
bindm = ... (mouse)

# Modifiers: SUPER, SHIFT, CTRL, ALT, MOD2, MOD3, MOD4, MOD5
# Combine with spaces: SUPER SHIFT

## DISPATCHERS
# Window Management
dispatch killactive          Close active window
dispatch togglefloating      Toggle floating/tiling
dispatch fullscreen          Toggle fullscreen (0=toggle, 1=full width)
dispatch pseudo              Toggle pseudotile
dispatch pin                 Pin window to all workspaces
dispatch centerwindow        Center floating window

# Focus
dispatch movefocus           Move focus (l/r/u/d)
dispatch cyclenext           Cycle focus to next window
dispatch focuswindow         Focus specific window

# Movement
dispatch movewindow          Move window (l/r/u/d)
dispatch swapwindow          Swap with window (l/r/u/d)
dispatch movetoworkspace     Move to workspace ID
dispatch resizeactive        Resize active window

# Workspaces
dispatch workspace           Switch to workspace
dispatch togglespecialworkspace Toggle special workspace

# Groups
dispatch togglegroup         Toggle window grouping
dispatch changegroupactive   Cycle group windows (f/b)
dispatch moveintogroup       Join group (l/r/u/d)
dispatch moveoutofgroup      Leave group

# Other
dispatch exec                Execute command
dispatch pass                Pass to other handler
dispatch layoutmsg           Send layout message

## COMMON PATTERNS
# Lock screen
dispatch exec loginctl lock-session

# Reload config
dispatch exec hyprctl reload

# App launchers
dispatch exec uwsm-app -- APP_NAME

## FILE LOCATIONS
Main config:    ~/.config/hypr/hyprland.conf
Keybindings:    ~/.config/hypr/bindings/
Window rules:   ~/.config/hypr/apps/
Autostart:      ~/.config/hypr/autostart.conf
Logs:           /tmp/hypr/*/hyprland.log

## ENVIRONMENT VARIABLES
XCURSOR_SIZE                  Cursor size for XWayland
HYPRCURSOR_SIZE               Cursor size for Wayland
GDK_SCALE                     GTK app scaling
QT_SCALE_FACTOR               Qt app scaling
XDG_CURRENT_DESKTOP           Desktop environment name
MOZ_ENABLE_WAYLAND            Enable Wayland for Firefox
QT_QPA_PLATFORM               Qt platform (wayland;xcb)

## USEFUL HYPRCTL COMMANDS
hyprctl reload               Reload configuration
hyprctl monitors             List monitors
hyprctl workspaces           List workspaces
hyprctl clients              List windows/clients
hyprctl activewindow         Show active window info
hyprctl binds                List all keybindings
hyprctl dispatch             Execute dispatcher
hyprctl keyword              Set config variable
hyprctl getoption            Get config value
hyprctl version              Show version info
hyprctl systeminfo           Show system information
